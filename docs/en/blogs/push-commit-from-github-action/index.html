name: example
on: push
jobs:
  example:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v2
    - run: |
        git config --global user.name '<user.name>'
        git config --global user.email '<user.email>'
        git add .
        git commit -snm "<commit message>" || true
        git push origin <branch> || true
