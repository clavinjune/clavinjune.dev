package main

import (
	"bufio"
	"encoding/base64"
	"log"
	"os"
)

func main() {
	l := log.New(os.Stdout, "[STREAM] ", log.Lshortfile|log.LstdFlags|log.Lmsgprefix)
	s := bufio.NewScanner(os.Stdin)

	for s.Scan() {
		t := s.Text()
		b := s.Bytes()

		// for example process the input to b64
		b64 := base64.StdEncoding.EncodeToString(b)
		l.Println(t, "=>", b64)
	}
}
